 25 systemctl_redirect () {
 26     local s
 27     local prog=${1##*/}
 28     local command=$2
 29     local options=""
 30 
 31     case "$command" in
 32     start)
 33         s=$"Starting $prog (via systemctl): "
 34         ;;
 35     stop)
 36         s=$"Stopping $prog (via systemctl): "
 37         ;;
 38     reload|try-reload)
 39         s=$"Reloading $prog configuration (via systemctl): "
 40         ;;
 41     restart|try-restart|condrestart)
 42         s=$"Restarting $prog (via systemctl): "
 43         ;;
 44     esac
 45 
 46     if [ -n "$SYSTEMCTL_IGNORE_DEPENDENCIES" ] ; then
 47         options="--ignore-dependencies"
 48     fi
 49 
 50     if ! systemctl show "$prog.service" > /dev/null 2>&1 || \
 51             systemctl show -p LoadState "$prog.service" | grep -q 'not-found' ; then
 52         action $"Reloading systemd: " /bin/systemctl daemon-reload
 53     fi
 54 
 55     action "$s" /bin/systemctl $options $command "$prog.service"
 56 }

102 # Check if any of $pid (could be plural) are running
103 checkpid() {
104     local i
105 
106     for i in $* ; do
107         [ -d "/proc/$i" ] && return 0
108     done
109     return 1
110 }

112 __kill_pids_term_kill_checkpids() {
113     local base_stime=$1
114     shift 1
115     local pid=
116     local pids=$*
117     local remaining=
118     local stat=
119     local stime=
120 
121     for pid in $pids ; do
122         [ ! -e  "/proc/$pid" ] && continue
123         read -r line < "/proc/$pid/stat" 2> /dev/null
124 
125         stat=($line)
126         stime=${stat[21]}
127 
128         [ -n "$stime" ] && [ "$base_stime" -lt "$stime" ] && continue
129         remaining+="$pid "
130     done
131 
132     echo "$remaining"
133     [ -n "$remaining" ] && return 1
134 
135     return 0
136 }

138 __kill_pids_term_kill() {
139     local try=0
140     local delay=3;
141     local pid=
142     local stat=($(< /proc/self/stat))
143     local base_stime=${stat[21]}
144 
145     if [ "$1" = "-d" ]; then
146         delay=$2
147         shift 2
148     fi
149 
150     local kill_list=$*
151 
152     kill_list=$(__kill_pids_term_kill_checkpids $base_stime $kill_list)
153 
154     [ -z "$kill_list" ] && return 0
155 
156     kill -TERM $kill_list >/dev/null 2>&1
157     usleep 100000
158 
159     kill_list=$(__kill_pids_term_kill_checkpids $base_stime $kill_list)
160     if [ -n "$kill_list" ] ; then
161         while [ $try -lt $delay ] ; do
162             sleep 1
163             kill_list=$(__kill_pids_term_kill_checkpids $base_stime $kill_list)
164             [ -z "$kill_list" ] && break
165             let try+=1
166         done
167         if [ -n "$kill_list" ] ; then
168             kill -KILL $kill_list >/dev/null 2>&1
169             usleep 100000
170             kill_list=$(__kill_pids_term_kill_checkpids $base_stime $kill_list)
171         fi
172     fi
173 
174     [ -n "$kill_list" ] && return 1
175     return 0
176 }

178 # __proc_pids {program} [pidfile]
179 # Set $pid to pids from /var/run* for {program}.  $pid should be declared
180 # local in the caller.
181 # Returns LSB exit code for the 'status' action.
182 __pids_var_run() {
183     local base=${1##*/}
184     local pid_file=${2:-/var/run/$base.pid}
185     local pid_dir=$(/usr/bin/dirname $pid_file > /dev/null)
186     local binary=$3
187 
188     [ -d "$pid_dir" -a ! -r "$pid_dir" ] && return 4
189 
190     pid=
191     if [ -f "$pid_file" ] ; then
192             local line p
193 
194         [ ! -r "$pid_file" ] && return 4 # "user had insufficient privilege"
195         while : ; do
196             read line
197             [ -z "$line" ] && break
198             for p in $line ; do
199                 if [ -z "${p//[0-9]/}" ] && [ -d "/proc/$p" ] ; then
200                     if [ -n "$binary" ] ; then
201                         local b=$(readlink /proc/$p/exe | sed -e 's/\s*(deleted)$//')
202                         [ "$b" != "$binary" ] && continue
203                     fi
204                     pid="$pid $p"
205                 fi
206             done
207         done < "$pid_file"
208 
209             if [ -n "$pid" ]; then
210                     return 0
211             fi
212         return 1 # "Program is dead and /var/run pid file exists"
213     fi
214     return 3 # "Program is not running"
215 }

217 # Output PIDs of matching processes, found using pidof
218 __pids_pidof() {
219     pidof -c -m -o $$ -o $PPID -o %PPID -x "$1" || \
220         pidof -c -m -o $$ -o $PPID -o %PPID -x "${1##*/}"
221 }

224 # A function to start a program.
225 daemon() {
226     # Test syntax.
227     local gotbase= force= nicelevel corelimit
228     local pid base= user= nice= bg= pid_file=
229     local cgroup=
230     nicelevel=0
231     while [ "$1" != "${1##[-+]}" ]; do
232         case $1 in
233         '')
234             echo $"$0: Usage: daemon [+/-nicelevel] {program}" "[arg1]..."
235             return 1
236             ;;
237         --check)
238             base=$2
239             gotbase="yes"
240             shift 2
241             ;;
242         --check=?*)
243             base=${1#--check=}
244             gotbase="yes"
245             shift
246             ;;
247         --user)
248             user=$2
249             shift 2
250             ;;
251         --user=?*)
252             user=${1#--user=}
253             shift
254             ;;
255         --pidfile)
256             pid_file=$2
257             shift 2
258             ;;
259         --pidfile=?*)
260             pid_file=${1#--pidfile=}
261             shift
262             ;;
263         --force)
264             force="force"
265             shift
266             ;;
267         [-+][0-9]*)
268             nice="nice -n $1"
269             shift
270             ;;
271         *)
272             echo $"$0: Usage: daemon [+/-nicelevel] {program}" "[arg1]..."
273             return 1
274             ;;
275       esac
276     done
277 
278     # Save basename.
279     [ -z "$gotbase" ] && base=${1##*/}
280 
281     # See if it's already running. Look *only* at the pid file.
282     __pids_var_run "$base" "$pid_file"
283 
284     [ -n "$pid" -a -z "$force" ] && return
285 
286     # make sure it doesn't core dump anywhere unless requested
287     corelimit="ulimit -S -c ${DAEMON_COREFILE_LIMIT:-0}"
288 
289     # if they set NICELEVEL in /etc/sysconfig/foo, honor it
290     [ -n "${NICELEVEL:-}" ] && nice="nice -n $NICELEVEL"
291 
292     # if they set CGROUP_DAEMON in /etc/sysconfig/foo, honor it
293     if [ -n "${CGROUP_DAEMON}" ]; then
294         if [ ! -x /bin/cgexec ]; then
295             echo -n "Cgroups not installed"; warning
296             echo
297         else
298             cgroup="/bin/cgexec";
299             for i in $CGROUP_DAEMON; do
300                 cgroup="$cgroup -g $i";
301             done
302         fi
303     fi
304 
305     # Echo daemon
306     [ "${BOOTUP:-}" = "verbose" -a -z "${LSB:-}" ] && echo -n " $base"
307 
308     # And start it up.
309     if [ -z "$user" ]; then
310        $cgroup $nice /bin/bash -c "$corelimit >/dev/null 2>&1 ; $*"
311     else
312        $cgroup $nice runuser -s /bin/bash $user -c "$corelimit >/dev/null 2>&1 ; $*"
313     fi
314 
315     [ "$?" -eq 0 ] && success $"$base startup" || failure $"$base startup"
316 }

318 # A function to stop a program.
319 killproc() {
320     local RC killlevel= base pid pid_file= delay try binary=
321 
322     RC=0; delay=3; try=0
323     # Test syntax.
324     if [ "$#" -eq 0 ]; then
325         echo $"Usage: killproc [-p pidfile] [ -d delay] {program} [-signal]"
326         return 1
327     fi
328     if [ "$1" = "-p" ]; then
329         pid_file=$2
330         shift 2
331     fi
332     if [ "$1" = "-b" ]; then
333         if [ -z $pid_file ]; then
334             echo $"-b option can be used only with -p"
335             echo $"Usage: killproc -p pidfile -b binary program"
336             return 1
337         fi
338         binary=$2
339         shift 2
340     fi
341     if [ "$1" = "-d" ]; then
342         delay=$(echo $2 | awk -v RS=' ' -v IGNORECASE=1 '{if($1!~/^[0-9.]+[smhd]?$/) exit 1;d=$1~/s$|^[0-9.]*$/?1:$1~/m$/?60:$1~/h$/?60*60:$1~/d$/?24*60*60:-1;if(d==-1) exit 1;delay+=d*$1} END {printf("%d",delay+0.5)}')
343         if [ "$?" -eq 1 ]; then
344             echo $"Usage: killproc [-p pidfile] [ -d delay] {program} [-signal]"
345             return 1
346         fi
347         shift 2
348     fi
349 
350 
351     # check for second arg to be kill level
352     [ -n "${2:-}" ] && killlevel=$2
353 
354     # Save basename.
355     base=${1##*/}
356 
357     # Find pid.
358     __pids_var_run "$1" "$pid_file" "$binary"
359     RC=$?
360     if [ -z "$pid" ]; then
361         if [ -z "$pid_file" ]; then
362             pid="$(__pids_pidof "$1")"
363         else
364             [ "$RC" = "4" ] && { failure $"$base shutdown" ; return $RC ;}
365         fi
366     fi
367 
368     # Kill it.
369     if [ -n "$pid" ] ; then
370         [ "$BOOTUP" = "verbose" -a -z "${LSB:-}" ] && echo -n "$base "
371         if [ -z "$killlevel" ] ; then
372             __kill_pids_term_kill -d $delay $pid
373             RC=$?
374             [ "$RC" -eq 0 ] && success $"$base shutdown" || failure $"$base shutdown"
375         # use specified level only
376         else
377             if checkpid $pid; then
378                 kill $killlevel $pid >/dev/null 2>&1
379                 RC=$?
380                 [ "$RC" -eq 0 ] && success $"$base $killlevel" || failure $"$base $killlevel"
381             elif [ -n "${LSB:-}" ]; then
382                 RC=7 # Program is not running
383             fi
384         fi
385     else
386         if [ -n "${LSB:-}" -a -n "$killlevel" ]; then
387             RC=7 # Program is not running
388         else
389             failure $"$base shutdown"
390             RC=0
391         fi
392     fi
393 
394     # Remove pid file if any.
395     if [ -z "$killlevel" ]; then
396         rm -f "${pid_file:-/var/run/$base.pid}"
397     fi
398     return $RC
399 }

416 # A function to find the pid of a program.
417 pidofproc() {
418     local RC pid pid_file=
419 
420     # Test syntax.
421     if [ "$#" = 0 ]; then
422         echo $"Usage: pidofproc [-p pidfile] {program}"
423         return 1
424     fi
425     if [ "$1" = "-p" ]; then
426         pid_file=$2
427         shift 2
428     fi
429     fail_code=3 # "Program is not running"
430 
431     # First try "/var/run/*.pid" files
432     __pids_var_run "$1" "$pid_file"
433     RC=$?
434     if [ -n "$pid" ]; then
435         echo $pid
436         return 0
437     fi
438 
439     [ -n "$pid_file" ] && return $RC
440     __pids_pidof "$1" || return $RC
441 }

443 status() {
444     local base pid lock_file= pid_file= binary=
445 
446     # Test syntax.
447     if [ "$#" = 0 ] ; then
448         echo $"Usage: status [-p pidfile] {program}"
449         return 1
450     fi
451     if [ "$1" = "-p" ]; then
452         pid_file=$2
453         shift 2
454     fi
455     if [ "$1" = "-l" ]; then
456         lock_file=$2
457         shift 2
458     fi
459     if [ "$1" = "-b" ]; then
460         if [ -z $pid_file ]; then
461             echo $"-b option can be used only with -p"
462             echo $"Usage: status -p pidfile -b binary program"
463             return 1
464         fi
465         binary=$2
466         shift 2
467     fi
468     base=${1##*/}
469 
470     if [ "$_use_systemctl" = "1" ]; then
471         systemctl status ${0##*/}.service
472         ret=$?
473         # LSB daemons that dies abnormally in systemd looks alive in systemd's eyes due to RemainAfterExit=yes
474         # lets adjust the reality a little bit
475         if systemctl show -p ActiveState ${0##*/}.service | grep -q '=active$' && \
476         systemctl show -p SubState ${0##*/}.service | grep -q '=exited$' ; then
477             ret=3
478         fi
479         return $ret
480     fi
481 
482     # First try "pidof"
483     __pids_var_run "$1" "$pid_file" "$binary"
484     RC=$?
485     if [ -z "$pid_file" -a -z "$pid" ]; then
486         pid="$(__pids_pidof "$1")"
487     fi
488     if [ -n "$pid" ]; then
489         echo $"${base} (pid $pid) is running..."
490         return 0
491     fi
492 
493     case "$RC" in
494     0)
495         echo $"${base} (pid $pid) is running..."
496         return 0
497         ;;
498     1)
499         echo $"${base} dead but pid file exists"
500         return 1
501         ;;
502     4)
503         echo $"${base} status unknown due to insufficient privileges."
504         return 4
505         ;;
506     esac
507     if [ -z "${lock_file}" ]; then
508         lock_file=${base}
509     fi
510     # See if /var/lock/subsys/${lock_file} exists
511     if [ -f /var/lock/subsys/${lock_file} ]; then
512         echo $"${base} dead but subsys locked"
513         return 2
514     fi
515     echo $"${base} is stopped"
516     return 3
517 }

648 # Apply sysctl settings, including files in /etc/sysctl.d
649 apply_sysctl() {
650     if [ -x /lib/systemd/systemd-sysctl ]; then
651     /lib/systemd/systemd-sysctl
652     else
653         for file in /usr/lib/sysctl.d/*.conf ; do
654             is_ignored_file "$file" && continue
655             [ -f /run/sysctl.d/${file##*/} ] && continue
656             [ -f /etc/sysctl.d/${file##*/} ] && continue
657             test -f "$file" && sysctl -e -p "$file" >/dev/null 2>&1
658         done
659         for file in /run/sysctl.d/*.conf ; do
660             is_ignored_file "$file" && continue
661             [ -f /etc/sysctl.d/${file##*/} ] && continue
662             test -f "$file" && sysctl -e -p "$file" >/dev/null 2>&1
663         done
664         for file in /etc/sysctl.d/*.conf ; do
665             is_ignored_file "$file" && continue
666             test -f "$file" && sysctl -e -p "$file" >/dev/null 2>&1
667         done
668         sysctl -e -p /etc/sysctl.conf >/dev/null 2>&1
669     fi
670 }
